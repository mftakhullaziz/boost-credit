### Makefile Helper
	
	# RUNNING CLEAN TEST
	make cleanTest/bookService
		
	# RUNNING CLEAN BUILD
	make clean-build/bookService
		
	# RUNNING BUILD
	make build/bookService
		
	# RUNNING SERVICE
	make run/bookService
		
	# RUNNING BUILD AND RUN SERVICE
	make service/bookService
